//fixme events from here onwards 
 An event is a member (not a type) built on top of a delegate that provides a controlled, publish-subscribe mechanism. 
 It restricts how the underlying delegate can be used from outside the declaring class: external code can only += (subscribe) or -= (unsubscribe), but cannot directly invoke the delegate or overwrite its invocation list (=), and cannot see/clear other subscribers' handlers.
 Essentially: an event is a delegate with access-restricted invocation, following the observer/publish-subscribe design pattern.
    class Publisher
    {
        // Declares an event based on a delegate type
        public event EventHandler<string> OnMessage;

        public void RaiseMessage(string msg)
        {
            // Only the declaring class can invoke it
            OnMessage?.Invoke(this, msg);
        }
    }

    class Subscriber
    {
        public void Handle(object sender, string msg) =>
            Console.WriteLine($"Received: {msg}");
    }

    var pub = new Publisher();
    var sub = new Subscriber();

    pub.OnMessage += sub.Handle;  // ✅ subscribing is allowed
    // pub.OnMessage(pub, "hi"); // ❌ compile error — can't invoke from outside
    // pub.OnMessage = null;     // ❌ compile error — can't overwrite from outside

    pub.RaiseMessage("Hello!");   // triggers all subscribers
