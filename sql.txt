1. Create a user defined function fn_annual_salary(emp_id INT) that returns the annual
salary of an employee by calculating (SAL × 12) + (COMM × 12), treating NULL
commission as 0.
    DELIMITER //

    CREATE FUNCTION fn_annual_salary(emp_id INT)
    RETURNS INT
    DETERMINISTIC
    BEGIN
        DECLARE salary INT;
        DECLARE commission INT;

        SELECT SAL, IFNULL(COMM, 0)
        INTO salary, commission
        FROM My_EMP
        WHERE EMPNO = emp_id;

        RETURN (salary * 12) + (commission * 12);
    END //

    DELIMITER ;

5. Create a function fn_salary_vs_dept_avg(emp_id INT) that compares an employee’s
salary with the average salary of their department and returns 'Above Avg', 'Below
Avg', or 'Equal Avg' accordingly.
    DELIMITER //

    CREATE FUNCTION fn_salary_vs_dept_avg(emp_id INT)
    RETURNS VARCHAR(20)
    DETERMINISTIC
    BEGIN
        DECLARE emp_sal INT;
        DECLARE dept_avg DECIMAL(10,2);
        DECLARE dept_no INT;

        SELECT SAL, DEPTNO
        INTO emp_sal, dept_no
        FROM My_EMP
        WHERE EMPNO = emp_id;

        SELECT AVG(SAL)
        INTO dept_avg
        FROM My_EMP
        WHERE DEPTNO = dept_no;

        IF emp_sal > dept_avg THEN
            RETURN 'Above Avg';
        ELSEIF emp_sal < dept_avg THEN
            RETURN 'Below Avg';
        ELSE
            RETURN 'Equal Avg';
        END IF;

    END //

    DELIMITER ;
